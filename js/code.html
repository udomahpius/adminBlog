text = input("Enter a text: ") # Initialize a dictionary to store vowel counts vowels = {'a': 0, 'e': 0, 'i': 0, 'o': 0, 'u': 0} # Convert text to lowercase to handle both upper and lower cases text = text.lower() # Loop through each character in the text for char in text: if char in vowels: # Check if the character is a vowel vowels[char] += 1 # Print the frequency of each vowel print("\nVowel Frequencies:") for vowel, count in vowels.items(): print(f"{vowel}: {count}")
